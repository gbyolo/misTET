Object.extend(Object, {

    getsID: function (id) {
                
        var array = $A(this.getElementsByTagName('*'));
        var result = new Array();
                        
	    for (var i = 0; i < array.length; i++) {
			if (array[i].getAttribute('id') == id) {
				result.push(array[i]);
			}
	    }
                        
	    if (result.length != 0) {
			return result;
	    } else {
			return false;
	    }
                
    }
        
});
